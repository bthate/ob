# basic set

>>> from ob.obj import Object
>>> o = Object()
>>> o.set("key", "value")
>>> o.key
'value'

# attribute set

>>> from ob.obj import Object
>>> o = Object()
>>> o.key = "value"
>>> o.key
'value'

# save/load

>>> from ob.krn import k
>>> k.cfg.wd = ".test"
>>> from ob.obj import Object
>>> o = Object()
>>> o["key"] = "value"
>>> p = o.save()
>>> p
'ob.obj.Object/...
>>> oo = Object()
>>> oo.load(p)
{"key": "value"}
>>> oo.key
'value'
